string dir = "c:\\temp\\";

// تحميل صورة GIF متحركة

{
    // للضغط بدون فقد ، تؤدي زيادة إعداد الجودة إلى زيادة جودة الضغط وتقليل حجم الملف

    image.Save(

    image.Save(
        dir + "output_lossless_80.webp",

    image.Save(
        dir + "output_lossy_50.webp",

}

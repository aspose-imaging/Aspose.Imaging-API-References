```csharp
[C#]

string dir = "c:\\temp\\";

using (Aspose.Imaging.Image image = Aspose.Imaging.Image.Load(dir + "sample.gif"))

    saveOptions.Quality = 50;

    // حدد ما إذا كان يمكن حفظ الصورة في Jpeg
```
